# Bubble Sort — sıralama algoritması
def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        for j in range(0, n-i-1):
            if arr[j] > arr[j+1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]
    return arr

# Test
data = [64, 34, 25, 12, 22, 11, 90]
result = bubble_sort(data)
print("Sıralı:", result)
